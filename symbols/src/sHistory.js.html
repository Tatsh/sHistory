  2 /**
  3  * Page state management. Uses real hashcange event if possible. Partially
  4  *   based on work by Ben Alman.<br>
  5  * To compile: <code>closure-compiler --warning_level VERBOSE --compilation_level ADVANCED_OPTIMIZATIONS --js sHistory.js --js export.js --output_wrapper "(function(){%output%}())" > shistory.min.js</code>
  6  *
  7  * @see http://benalman.com/projects/jquery-bbq-plugin/
  8  * @constructor
  9  * @returns {sHistory} The history object.
 10  * @license sHistory (c) 2012 Andrew Udvare | http://www.opensource.org/licenses/mit-license.php
 11  */
 12 var sHistory = function () {
 13   if (!sHistory.hasNativeSupport) {
 14     // From ba-bbq
 15     /**
 16      * @param {string} [url]
 17      * @returns {string}
 18      * @private
 19      */
 20     var getFragment = function (url) {
 21       url = url || location.href;
 22       return '#' + url.replace(/^[^#]*#?(.*)$/, '$1');
 23     };
 24 
 25     // Create an iframe
 26     var iframe = document.createElement('iframe');
 27     iframe.setAttribute('tabindex', -1);
 28     iframe.setAttribute('title', 'empty');
 29     iframe.setAttribute('src', 'javascript:0');
 30     iframe.setAttribute('id', 'shistory');
 31     iframe.style.display = 'none';
 32 
 33     // The original hash
 34     var lastHash = getFragment();
 35 
 36     // Some callbacks that will be defined fully upon loading the iframe
 37     var fnRet = function (v) { return v; };
 38     var getHistory = fnRet;
 39     var setHistory = fnRet;
 40 
 41     var callEventHandlers = function () {
 42       for (var i = 0; i < sHistory._eventListeners.length; i++) {
 43         sHistory._eventListeners[i]();
 44       }
 45     };
 46 
 47     // The polling callback
 48     var poll = function () {
 49       var hash = getFragment();
 50       var historyHash = getHistory();
 51 
 52       if (hash !== lastHash) {
 53         lastHash = hash;
 54         setHistory(lastHash, historyHash);
 55 
 56         // Call the event handlers
 57         callEventHandlers();
 58       }
 59       else if (historyHash !== lastHash) {
 60         // First hash change
 61         location.href = location.href.replace(/#.*/, '') + historyHash;
 62       }
 63 
 64       setTimeout(poll, 50);
 65     };
 66     // Upon the iframe loading, make sure we begin polling for hash changes
 67     var loadOnceCallback = function () {
 68       var contentIframe = iframe.contentWindow;
 69 
 70       // Define set/getHistory
 71       getHistory = function () {
 72         return getFragment(contentIframe.location.href);
 73       };
 74       setHistory = function (hash, historyHash) {
 75         if (hash !== historyHash) {
 76           var iframeDoc = contentIframe.document;
 77           //iframeDoc.title = document.title;
 78           iframeDoc.open(); // This triggers a history event
 79           iframeDoc.close();
 80           contentIframe.location.hash = hash;
 81         }
 82       };
 83 
 84       // Make the iframe's hash the same as the page's
 85       contentIframe.document.open();
 86       contentIframe.location.hash = location.hash;
 87 
 88       // Begin polling
 89       poll();
 90 
 91       // This is always called from .start() so call the event handlers
 92       //   for first hash event
 93       callEventHandlers();
 94 
 95       // Unbind
 96       if (iframe.detachEvent) {
 97         iframe.detachEvent('onload', loadOnceCallback);
 98       }
 99       else if (iframe.removeEventListener) {
100         iframe.removeEventListener('load', loadOnceCallback);
101       }
102     };
103     // Bind the one-time callback
104     if (iframe.attachEvent) {
105       iframe.attachEvent('onload', loadOnceCallback);
106     }
107     else if (iframe.addEventListener) {
108       iframe.addEventListener('load', loadOnceCallback, false);
109     }
110 
111 
112     // Add the iframe to the page
113     document.body.appendChild(iframe);
114   }
115 
116   return this;
117 };
118 /**
119  * If the browser supports the <code>hashchange</code> event natively.
120  * @type boolean
121  */
122 sHistory.hasNativeSupport = (function () {
123   var mode = document.documentMode;
124   return 'onhashchange' in window && (mode === undefined || mode > 7);
125 })();

131 /**
132  * Get full URI.
133  * @private
134  * @returns {string}
135  */
136 sHistory._getFullURI = function () {
137   var url = location.protocol + '//' + location.hostname;
138   var qs = location.search ? location.search.substr(1) : '';
139 
140   if (qs) {
141     url += '?' + qs;
142   }
143 
144   return url;
145 };
146 /**
147  * Push a state. Note that setting an empty state can cause a browser to
148  *   scroll.
149  * @param {string} stateKey State name to push. If merge is <code>true</code>,
150  *   then this will replace the current state of the same key name. The
151  *   <code>__t</code> key is reserved.
152  * @param {string|boolean|number} stateValue Value to set.
153  * @param {boolean} [merge=true] Whether or not to merge with the current
154  *   state.
155  */
156 sHistory.pushState = function (stateKey, stateValue, merge) {
157   merge === undefined && (merge = true);
158 
159   if (stateKey === undefined || stateValue === undefined) {
160     return;
161   }
162 
163   if (stateKey === '__t') {
164     return;
165   }
166 
167   var url = sHistory._getFullURI();
168   var hash = location.hash.replace(/^#&/, '#');
169   var euc = encodeURIComponent;
170 
171   if (!hash) {
172     hash = '#';
173   }
174 
175   if (merge) {
176     if (sHistory.getState(stateKey) !== null) {
177       var keyValues = hash.substr(1).split('&'), split;
178       for (var i = 0; i < keyValues.length; i++) {
179         split = keyValues[i].split('=');
180         if (split[0] === stateKey) {
181           keyValues[i] = stateKey + '=' + euc(stateValue.toString());
182         }
183       }
184       hash = '#' + keyValues.join('&');
185     }
186     else {
187       if (location.hash) {
188         hash += '&';
189       }
190       hash += stateKey + '=' + euc(stateValue.toString());
191     }
192   }
193   else {
194     hash = '#' + stateKey + '=' + euc(stateValue.toString());
195   }
196 
197   location.href = url + hash;
198 };
199 /**
200  * Push states. Note that the <code>__t</code> key is reserved.
201  * @param {Object} stateObject Object of keys to values (string, number, or
202  *   boolean).
203  * @param {boolean} [merge=true] Whether or not to merge with the current
204  *   state.
205  */
206 sHistory.pushStates = function (stateObject, merge) {
207   merge === undefined && (merge = true);
208   stateObject === undefined && (stateObject = {});
209 
210   var euc = encodeURIComponent;
211   var url = sHistory._getFullURI();
212   var key, hash = '#';
213   var keyValues;
214 
215   if (!location.hash) {
216     merge = false;
217   }
218 
219   if (merge) {
220     // Compare the 2 objects, still need to keep order or 2 states would be pushed
221     // TODO Make this more efficient.
222     keyValues = location.hash.substr(1).split('&');
223     var found = false, split;
224 
225     for (key in stateObject) {
226       if (stateObject.hasOwnProperty(key)) {
227         if (sHistory.getState(key) !== null) {
228           for (var i = 0; i < keyValues.length; i++) {
229             split = keyValues[i].split('=');
230             if (split[0] === key) {
231               if (stateObject[key] === true) {
232                 stateObject[key] = 'true';
233               }
234               else if (stateObject[key] === false) {
235                 stateObject[key] = 'false';
236               }
237               keyValues[i] = key + '=' + euc(stateObject[key]);
238             }
239           }
240         }
241         else {
242           keyValues.push(key + '=' + euc(stateObject[key]));
243         }
244       }
245     }
246 
247     hash += keyValues.join('&');
248   }
249   else {
250     keyValues = [];
251 
252     for (key in stateObject) {
253       if (stateObject.hasOwnProperty(key)) {
254         // Reserved key
255         if (key === '__t') {
256           continue;
257         }
258 
259         if (stateObject[key] === true) {
260           stateObject[key] = 'true';
261         }
262         else if (stateObject[key] === false) {
263           stateObject[key] = 'false';
264         }
265 
266         keyValues.push(key + '=' + euc(stateObject[key]));
267       }
268     }
269 
270     hash = '#' + keyValues.join('&');
271   }
272 
273   location.href = url + hash;
274 };
275 /**
276  * Remove a state. If no arguments are passed, all states will be removed.
277  * @param {string} [stateName] The state to remove.
278  */
279 sHistory.removeState = function (stateName) {
280   if (stateName === undefined) {
281     location.href = sHistory._getFullURI() + '#';
282     return;
283   }
284 
285   if (sHistory.getState(stateName) !== null) {
286     // Get the entire hash and remove this state
287     var states = location.hash.substr(1).split('&');
288     var toPush = {};
289     for (var split, i = 0; i < states.length; i++) {
290       split = states[i].split('=');
291       if (split[0] === stateName) {
292         continue;
293       }
294 
295       toPush[split[0]] = decodeURIComponent(split[1]);
296     }
297 
298     var length = 0;
299     for (var key in toPush) {
300       if (toPush.hasOwnProperty(key)) {
301         length++;
302       }
303     }
304     if (length === 0) {
305       sHistory.removeState();
306     }
307 
308     sHistory.pushStates(toPush, false);
309   }
310 };
311 /**
312  * Get a state by key name. If no arguments are passed, returns whether or not
313  *   <code>location.hash</code> is empty.
314  * @param {string} [key] Key to use. Case-sensitive.
315  * @param {string} [castTo='string'] Cast to string, number (integer), float,
316  *   boolean.
317  * @param {string|number|boolean|null} [defaultValue] The default value to
318  *   return if the value does not exist.
319  * @returns {string|number|boolean|null} If <code>location.hash</code> is an
320  *   empty string, the state value requrested, the default value if specified,
321  *   or <code>null</code> if no such state exists.
322  */
323 sHistory.getState = function (key, castTo, defaultValue) {
324   if (arguments.length === 0) {
325     return !!location.hash;
326   }
327 
328   if (!key || key === '__t') {
329     return null;
330   }
331 
332   castTo === undefined && (castTo = 'string');
333 
334   var keyValues = location.hash.substr(1).split('&');
335   var split, ret = null, lcRet = null;
336   var trues = {
337     'true': 1,
338     'on': 1,
339     'yes': 1
340   };
341   var falses = {
342     'false': 1,
343     'no': 1,
344     'off': 1
345   };
346 
347   for (var i = 0; i < keyValues.length; i++) {
348     split = keyValues[i].split('=');
349     if (split[0] === key) {
350       ret = decodeURIComponent(split[1]);
351       lcRet = ret.toLowerCase();
352       break;
353     }
354   }
355 
356   if (ret === null && defaultValue !== undefined) {
357     return defaultValue;
358   }
359 
360   if (castTo !== 'string') {
361     switch (castTo) {
362       case 'number':
363       case 'integer':
364       case 'int':
365         ret = parseInt(ret, 10);
366         if (isNaN(ret)) {
367           ret = 0;
368         }
369         break;
370 
371       case 'float':
372         ret = parseFloat(ret);
373         break;
374 
375       case 'boolean':
376       case 'bool':
377         if (trues[lcRet] !== undefined) {
378           return true;
379         }
380         else if (falses[lcRet] !== undefined) {
381           return false;
382         }
383         return !!ret;
384     }
385   }
386 
387   return ret;
388 };
389 /**
390  * Triggered first event.
391  * @type boolean
392  * @private
393  */
394 sHistory._started = false;
395 /**
396  * Dispatches first hashchange event.
397  * @private
398  */
399 sHistory._dispatchFirst = function () {
400   var event;
401 
402   if (sHistory.hasNativeSupport) {
403     if (document.createEvent) {
404       event = document.createEvent('HTMLEvents');
405       event.initEvent('hashchange', true, false);
406       window.dispatchEvent(event);
407     }
408     else {
409       var hash = location.hash;
410       if (hash !== '') {
411         hash = hash.replace(/[\&\#]__t=[0-9]+\&?/, '');
412         hash += '&__t=' + (function () {
413           var alpha = '0123456789', output = '';
414           for (var i = 0; i < 10; i++) {
415             output += alpha.charAt(Math.floor(Math.random() * 9));
416           }
417           return output;
418         })();
419       }
420       location.hash = hash;
421     }
422   }
423   else {
424     sHistory();
425   }
426 };
427 /**
428  * Trigger the first <code>hashchange</code> event. Should be called once per
429  *   page and only after all listeners have been registered using
430  *   <code>sHistory.addEventListener</code>.
431  * @param {string} [defaultState] The default state key name.
432  * @param {string|number|boolean} [defaultStateValue] The default state value.
433  */
434 sHistory.start = function (defaultState, defaultStateValue) {
435   if (!sHistory._started) {
436     if (defaultState !== undefined &&
437         defaultStateValue !== undefined &&
438         sHistory.getState(defaultState) === null) {
439       // Simply trigger the first hash change with a value
440       // IE cannot do this, so call sHistory() to initialise the polling
441       if (!sHistory.hasNativeSupport) {
442         sHistory();
443       }
444 
445       sHistory.pushState(defaultState, defaultStateValue);
446     }
447     else {
448       sHistory._dispatchFirst();
449     }
450 
451     sHistory._started = true;
452   }
453 };
454 /**
455  * @private
456  * @type Array
457  */
458 sHistory._eventListeners = [];
459 /**
460  * Add an event listener. These are called every time the hash changes, which
461  *   means that if one key changes but another state stays the same, then all
462  *   listeners will still be called.
463  * <p>If you want to prevent the callback from running repetitively, save the
464  *   last state value outside of the function and make sure it is different
465  *   within the callback before the callback can continue.</p>
466  * @param {function()} func Callback.
467  */
468 sHistory.addEventListener = function (func) {
469   if (!sHistory.hasNativeSupport) {
470     sHistory._eventListeners.push(func);
471     return;
472   }
473 
474   if (window.addEventListener) {
475     window.addEventListener('hashchange', func, false);
476   }
477   else if (window.attachEvent) {
478     window.attachEvent('onhashchange', func);
479   }
480 };
